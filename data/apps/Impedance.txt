    // --- The JavaScript remains unchanged from the previous version ---
    (() => {
        // --- Universal resize handler ---
        const allCanvases = document.querySelectorAll('canvas');
        let resizeTimeout;
        window.addEventListener('resize', () => {
            clearTimeout(resizeTimeout);
            resizeTimeout = setTimeout(() => {
                allCanvases.forEach(canvas => {
                    const event = new CustomEvent('redrawCanvas');
                    canvas.dispatchEvent(event);
                });
            }, 100);
        });

        // --- Utility to size a canvas and its container ---
        function sizeCanvas(canvas) {
            const dpr = window.devicePixelRatio || 1;
            const parent = canvas.parentElement;
            const rect = parent.getBoundingClientRect();
            
            if (rect.width <= 0 || rect.height <= 0) {
                return { width: 0, height: 0 };
            }
            
            canvas.width = rect.width * dpr;
            canvas.height = rect.height * dpr;

            const ctx = canvas.getContext('2d');
            ctx.setTransform(dpr, 0, 0, dpr, 0, 0); // Use setTransform for robustness
            return { width: rect.width, height: rect.height };
        }

        // ===================================================================
        // ==                                                               ==
        // ==                   INITIALIZATION FUNCTIONS                    ==
        // ==                                                               ==
        // ===================================================================
        
        function initImpedanceSim() {
            const section = document.getElementById('impedance-simulator');
            if (!section) return;

            const canvas = section.querySelector('#impedanceCanvas');
            const ctx = canvas.getContext('2d');
            const ui = {
                pauseBtn: section.querySelector('#impedancePauseBtn'),
                resetBtn: section.querySelector('#impedanceResetBtn'),
                sliders: {
                    speed: section.querySelector('#impedanceSpeedSlider'),
                    amplitude: section.querySelector('#impedanceAmplitudeSlider'),
                    m1Density: section.querySelector('#m1DensitySlider'),
                    m1Stiffness: section.querySelector('#m1StiffnessSlider'),
                    m2Density: section.querySelector('#m2DensitySlider'),
                    m2Stiffness: section.querySelector('#m2StiffnessSlider'),
                },
                values: {
                    speed: section.querySelector('#impedanceSpeedValue'),
                    amplitude: section.querySelector('#impedanceAmplitudeValue'),
                    m1Density: section.querySelector('#m1DensityValue'),
                    m1Stiffness: section.querySelector('#m1StiffnessValue'),
                    m2Density: section.querySelector('#m2DensityValue'),
                    m2Stiffness: section.querySelector('#m2StiffnessValue'),
                },
            };
            let state = { isPaused: false, time: 0, animationFrameId: null, simulationSpeed: 0.3, amplitude: 150, m1: { density: 2, stiffness: 8 }, m2: { density: 8, stiffness: 8 } };
            const params = { c1: 1, c2: 1, R: 0, T: 1, pulseSpatialWidth1: 150, boundary_phys: 500 };
            let lastTime = 0;
            
            function animate(currentTime) {
                if (state.isPaused) return;
                if (!lastTime) lastTime = currentTime;
                const deltaTime = currentTime - lastTime;
                state.time += (deltaTime / 16.67) * state.simulationSpeed;
                lastTime = currentTime;
                draw();
                state.animationFrameId = requestAnimationFrame(animate);
            }
            function draw() {
                const { width, height } = sizeCanvas(canvas);
                if(width === 0) return;
                ctx.clearRect(0, 0, width, height);
                const centerY = Math.round(height / 2);
                const boundary_px = width / 2;
                drawGridAndBoundary(centerY, boundary_px, width, height);
                const resultantPath = new Path2D();
                let isFirstPoint = true;
                const scale = boundary_px / params.boundary_phys;
                const timeToBoundary = params.boundary_phys / params.c1;
                const incCenter = state.time * params.c1;
                const reflCenter = 2 * params.boundary_phys - state.time * params.c1;
                const transCenter = params.boundary_phys + (state.time - timeToBoundary) * params.c2;
                const width2 = params.pulseSpatialWidth1 * (params.c2 / params.c1);
                for (let x_px = 0; x_px <= width; x_px++) {
                    const x_phys = x_px / scale;
                    const y_i = waveFunction(x_phys, incCenter, params.pulseSpatialWidth1);
                    const y_r = waveFunction(x_phys, reflCenter, params.pulseSpatialWidth1) * params.R;
                    const y_t = waveFunction(x_phys, transCenter, width2) * params.T;
                    const y = (x_phys <= params.boundary_phys ? (y_i + y_r) : y_t) * state.amplitude;
                    const y_draw = centerY - y;
                    if (isFirstPoint) { resultantPath.moveTo(x_px, y_draw); isFirstPoint = false; }
                    else resultantPath.lineTo(x_px, y_draw);
                }
                ctx.strokeStyle = '#f59e0b'; ctx.lineWidth = 2.5; ctx.stroke(resultantPath);
            }
            function waveFunction(x, center, width) { const arg = x - center; return (Math.abs(arg) > width * 3.5) ? 0 : Math.exp(-Math.pow(arg / (width / 2), 2)); }
            function drawGridAndBoundary(cY, bX, w, h) {
                ctx.beginPath(); ctx.moveTo(0, cY); ctx.lineTo(w, cY);
                ctx.strokeStyle = '#e5e7eb'; ctx.lineWidth = 1; ctx.stroke();
                ctx.beginPath(); ctx.moveTo(bX, 0); ctx.lineTo(bX, h);
                ctx.strokeStyle = '#ef4444'; ctx.lineWidth = 2; ctx.stroke();
                ctx.font = '500 14px Inter'; ctx.fillStyle = '#6b7280'; ctx.textAlign = 'center';
                ctx.fillText('Medium 1', bX / 2, 30);
                ctx.fillText('Medium 2', bX * 1.5, 30);
            }
            function handlePlayPause() {
                state.isPaused = !state.isPaused;
                ui.pauseBtn.textContent = state.isPaused ? "Play" : "Pause";
                ui.pauseBtn.classList.toggle("sim-btn-action", !state.isPaused);
                ui.pauseBtn.classList.toggle("sim-btn-secondary", state.isPaused);
                if (!state.isPaused) { lastTime = performance.now(); animate(lastTime); }
            }
             function handleReset() {
                cancelAnimationFrame(state.animationFrameId);
                recalcPhysics();
                state.time = -params.pulseSpatialWidth1 * 3 / params.c1;
                lastTime = 0;
                // Start animation immediately on reset, unless it was already paused
                if (!state.isPaused) {
                    animate(performance.now());
                } else {
                    draw(); // Draw initial state if paused
                }
            }
            function updateControls() {
                const ampVal = parseFloat(ui.sliders.amplitude.value);
                state.amplitude = ((ampVal - 1) / 9) * (300 - 20) + 20;
                const sVal = parseFloat(ui.sliders.speed.value);
                state.simulationSpeed = sVal <= 5 ? ((sVal - 1) / 4) * (0.3 - 0.05) + 0.05 : ((sVal - 5) / 5) * (5.0 - 0.3) + 0.3;
                state.m1.density = parseFloat(ui.sliders.m1Density.value);
                state.m1.stiffness = parseFloat(ui.sliders.m1Stiffness.value);
                state.m2.density = parseFloat(ui.sliders.m2Density.value);
                state.m2.stiffness = parseFloat(ui.sliders.m2Stiffness.value);
                ui.values.speed.textContent = state.simulationSpeed.toFixed(2);
                ui.values.amplitude.textContent = state.amplitude.toFixed(0);
                ui.values.m1Density.textContent = ui.sliders.m1Density.value;
                ui.values.m1Stiffness.textContent = ui.sliders.m1Stiffness.value;
                ui.values.m2Density.textContent = ui.sliders.m2Density.value;
                ui.values.m2Stiffness.textContent = ui.sliders.m2Stiffness.value;
                recalcPhysics();
                if (state.isPaused) draw();
            }
            function recalcPhysics() {
                const { m1, m2 } = state; const eps = 1e-9;
                params.c1 = Math.sqrt(m1.stiffness / (m1.density + eps)) * 15;
                params.c2 = Math.sqrt(m2.stiffness / (m2.density + eps)) * 15;
                const Z1 = Math.sqrt(m1.stiffness * m1.density);
                const Z2 = Math.sqrt(m2.stiffness * m2.density);
                const denom = Z1 + Z2 + eps;
                params.R = (Z1 - Z2) / denom;
                params.T = (2 * Z1) / denom;
            }
            
            Object.values(ui.sliders).forEach(el => el.addEventListener("input", updateControls));
            ui.pauseBtn.addEventListener('click', handlePlayPause);
            ui.resetBtn.addEventListener('click', handleReset);
            updateControls(); // Set initial values from sliders
            handleReset(); // Initialize time and start animation (if not paused)
            canvas.addEventListener('redrawCanvas', draw);
        }

        // --- Master Initializer ---
        function initializeAll() {
            initImpedanceSim();
            // Trigger a resize event to make sure canvases are sized correctly on load
            window.dispatchEvent(new Event('resize'));
        }

        if (document.readyState === 'loading') {
            document.addEventListener('DOMContentLoaded', initializeAll);
        } else {
            initializeAll();
        }
        
    })();
