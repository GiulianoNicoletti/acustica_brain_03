    // --- JavaScript remains unchanged ---
    (() => {
        // --- Universal resize handler ---
        const allCanvases = document.querySelectorAll('canvas');
        let resizeTimeout;
        window.addEventListener('resize', () => {
            clearTimeout(resizeTimeout);
            resizeTimeout = setTimeout(() => {
                allCanvases.forEach(canvas => {
                    const event = new CustomEvent('redrawCanvas');
                    canvas.dispatchEvent(event);
                });
            }, 100);
        });

        // --- Utility to size a canvas and its container ---
        function sizeCanvas(canvas) {
            const dpr = window.devicePixelRatio || 1;
            const parent = canvas.parentElement;
            const rect = parent.getBoundingClientRect();
            
            if (rect.width <= 0 || rect.height <= 0) {
                return { width: 0, height: 0 };
            }
            
            canvas.width = rect.width * dpr;
            canvas.height = rect.height * dpr;

            const ctx = canvas.getContext('2d');
            ctx.setTransform(dpr, 0, 0, dpr, 0, 0); // Use setTransform for robustness
            return { width: rect.width, height: rect.height };
        }

        // ===================================================================
        // ==                                                               ==
        // ==                   INITIALIZATION FUNCTIONS                    ==
        // ==                                                               ==
        // ===================================================================

        // Removed initHarmonicSim and initImpedanceSim as they are not in this file's HTML

        function initStandingWaveSim() {
            const section = document.getElementById('standing-wave-simulator');
            if (!section) return;

            const canvas = section.querySelector('#standingWaveCanvas');
            const ctx = canvas.getContext('2d');
            const pauseBtn = section.querySelector('#standingWavePauseBtn');
            const resetBtn = section.querySelector('#standingWaveResetBtn');
            const viewLengthSlider = section.querySelector('#viewLengthSlider');
            const viewLengthValue = section.querySelector('#viewLengthValue');
            const viewModeBtns = section.querySelector('#viewModeBtns');
            let state = { isPaused: false, time: 0, animationFrameId: null, viewMode: 'dipole', showIncident: true, showReflected: true, showResultant: true };
            const params = { amplitude: 80, speed: 300, wavelength: 800, viewLength: 800, get k() { return 2 * Math.PI / this.wavelength; }, get period() { return this.wavelength / this.speed; }, get omega() { return 2 * Math.PI / this.period; } };
            let lastTime = 0;

            function animate(currentTime) {
                if (state.isPaused) return;
                if (!lastTime) lastTime = currentTime;
                const deltaTime = (currentTime - lastTime) / 1000;
                state.time += deltaTime;
                lastTime = currentTime;
                draw();
                state.animationFrameId = requestAnimationFrame(animate);
            }
            function draw() {
                const { width, height } = sizeCanvas(canvas);
                if(width === 0) return;
                const L_px = width;
                const L_phys = params.viewLength;
                const scale = L_px / L_phys;
                const centerY = Math.round(height / 2);
                ctx.clearRect(0, 0, L_px, height);
                drawGrid(centerY, L_px);
                drawBoundary(L_px, centerY);
                const reflectionTime = L_phys / params.speed;
                const incidentFront = params.speed * state.time;
                const reflectedFront = L_phys - params.speed * (state.time - reflectionTime);
                const paths = { incident: new Path2D(), reflected: new Path2D(), resultant: new Path2D() };
                let firstPoints = { incident: true, reflected: true, resultant: true };
                for (let x_px = 0; x_px <= L_px; x_px++) {
                    const x_phys = x_px / scale;
                    let y_i = 0, y_r = 0;
                    if (x_phys <= incidentFront) y_i = params.amplitude * Math.sin(params.k * x_phys - params.omega * state.time);
                    if (state.time >= reflectionTime && x_phys >= reflectedFront) {
                        y_r = -1 * params.amplitude * Math.sin(params.k * (2 * L_phys - x_phys) - params.omega * state.time);
                        if (state.showReflected) {
                            if (firstPoints.reflected) { paths.reflected.moveTo(x_px, centerY - y_r); firstPoints.reflected = false; }
                            else { paths.reflected.lineTo(x_px, centerY - y_r); }
                        }
                    }
                    const y_total = y_i + y_r;
                    if (state.showIncident) {
                        if (firstPoints.incident) { paths.incident.moveTo(x_px, centerY - y_i); firstPoints.incident = false; }
                        else { paths.incident.lineTo(x_px, centerY - y_i); }
                    }
                    if (state.showResultant) {
                        if (firstPoints.resultant) { paths.resultant.moveTo(x_px, centerY - y_total); firstPoints.resultant = false; }
                        else { paths.resultant.lineTo(x_px, centerY - y_total); }
                        if (state.time >= reflectionTime && x_phys >= reflectedFront && x_px % 15 === 0) drawOscillator(x_px, centerY, y_total);
                    }
                }
                if (state.showIncident) { ctx.strokeStyle = '#3b82f6'; ctx.lineWidth = 1.5; ctx.stroke(paths.incident); }
                if (state.showReflected) { ctx.strokeStyle = '#10b981'; ctx.lineWidth = 1.5; ctx.stroke(paths.reflected); }
                if (state.showResultant) { ctx.strokeStyle = '#f59e0b'; ctx.lineWidth = 2.5; ctx.stroke(paths.resultant); }
            }
            function drawOscillator(x, cY, y) { ctx.beginPath(); ctx.moveTo(x, cY); ctx.lineTo(x, cY - y); ctx.strokeStyle = 'rgba(245, 158, 11, 0.5)'; ctx.lineWidth = 1; ctx.stroke(); }
            function drawGrid(cY, w) { ctx.beginPath(); ctx.moveTo(0, cY); ctx.lineTo(w, cY); ctx.strokeStyle = '#e5e7eb'; ctx.lineWidth = 1; ctx.stroke(); }
            function drawBoundary(L, cY) { ctx.beginPath(); ctx.moveTo(L, cY - params.amplitude - 20); ctx.lineTo(L, cY + params.amplitude + 20); ctx.strokeStyle = '#ef4444'; ctx.lineWidth = 2; ctx.stroke(); }
            function handlePlayPause() {
                state.isPaused = !state.isPaused;
                pauseBtn.textContent = state.isPaused ? 'Play' : 'Pause';
                pauseBtn.classList.toggle('sim-btn-action', !state.isPaused);
                pauseBtn.classList.toggle('sim-btn-secondary', state.isPaused);
                if (!state.isPaused) { lastTime = performance.now(); animate(lastTime); }
            }
            function handleReset() {
                cancelAnimationFrame(state.animationFrameId);
                state.time = 0; lastTime = 0;
                 // Start animation immediately on reset, unless it was already paused
                if (!state.isPaused) {
                    animate(performance.now());
                } else {
                    draw(); // Draw initial state if paused
                }
            }
            function handleViewModeChange(event) {
                const selected = event.target.closest('.view-mode-btn');
                if (!selected) return;
                state.viewMode = selected.dataset.mode;
                section.querySelectorAll('.view-mode-btn').forEach(b => b.classList.remove('view-mode-btn-active'));
                selected.classList.add('view-mode-btn-active');
                if (state.viewMode === 'full') {
                    viewLengthSlider.disabled = false;
                } else {
                    viewLengthSlider.disabled = true;
                    if (state.viewMode === 'monopole') params.viewLength = params.wavelength / 2;
                    else if (state.viewMode === 'dipole') params.viewLength = params.wavelength;
                    else if (state.viewMode === 'tripole') params.viewLength = params.wavelength * 1.5;
                    viewLengthSlider.value = params.viewLength;
                    viewLengthValue.textContent = Math.round(params.viewLength);
                }
                if(state.isPaused) draw();
            }
            function updateFromControls() {
                state.showIncident = section.querySelector('#showIncident').checked;
                state.showReflected = section.querySelector('#showReflected').checked;
                state.showResultant = section.querySelector('#showResultant').checked;
                if (!viewLengthSlider.disabled) {
                     params.viewLength = parseInt(viewLengthSlider.value, 10);
                     viewLengthValue.textContent = params.viewLength;
                }
                if(state.isPaused) draw();
            }
            
            section.querySelectorAll('input[type="checkbox"]').forEach(el => el.addEventListener('change', updateFromControls));
            viewLengthSlider.addEventListener('input', updateFromControls);
            pauseBtn.addEventListener('click', handlePlayPause);
            resetBtn.addEventListener('click', handleReset);
            viewModeBtns.addEventListener('click', handleViewModeChange);
            const initialModeBtn = section.querySelector('.view-mode-btn[data-mode="dipole"]');
            if (initialModeBtn) initialModeBtn.click(); // Set initial mode
            updateFromControls(); // Set initial checkbox states
            handleReset(); // Initialize time and start animation (if not paused)
            canvas.addEventListener('redrawCanvas', draw);
        }

        // --- Master Initializer ---
        function initializeAll() {
            // Only initialize the standing wave simulator
            initStandingWaveSim();
            // Trigger a resize event to make sure canvases are sized correctly on load
            window.dispatchEvent(new Event('resize'));
        }

        if (document.readyState === 'loading') {
            document.addEventListener('DOMContentLoaded', initializeAll);
        } else {
            initializeAll();
        }
        
    })();
